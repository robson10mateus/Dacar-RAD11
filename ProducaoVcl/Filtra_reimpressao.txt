SELECT
  A.ID_IDENREGIPROD,
  A.ID_PRODMATEEMBA,
  A.QN_PESOLIQUREGIPROD,
  A.QN_PESOPADRREGIPROD,
  A.QN_TARAEMBAREGIPROD,
  A.DT_BALAREGIPROD,
  B.FL_PADRAO,
  B.NM_PRODREDUMATEEMBA,
  A.DIGITOVER
FROM
  PRODUCAO_REGISTRO A,
  MATERIAL_EMBALAGEM B
WHERE
  A.ID_PRODMATEEMBA = B.ID_PRODMATEEMBA
   AND A.ID_IDENREGIPROD >= '160211050001' AND A.ID_IDENREGIPROD <= '160211050001' AND A.FL_STATREGIPROD = 'ES'
ORDER BY
  A.ID_IDENREGIPROD,
  A.ID_PRODMATEEMBA
